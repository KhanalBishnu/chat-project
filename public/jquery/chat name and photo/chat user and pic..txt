use Illuminate\Support\Facades\View;

public function showGroupChat($groupId)
{
    $groupChatMessages = GroupChat::with('sender')->where('group_id', $groupId)->get();

    return view('group_chat.show', compact('groupChatMessages'));
}

public function getProfilePictureUrlAttribute()
{
    $profilePicture = $this->getMedia('profile_picture')->first();

    if ($profilePicture) {
        return $profilePicture->getFullUrl();
    }

    return null;
}
