use Intervention\Image\Facades\Image;

// Load the image from the storage
$image = Image::make(storage_path('app/public/images/sample.jpg'));

// Resize the image to a specific width and maintain the aspect ratio
$image->resize(800, null, function ($constraint) {
    $constraint->aspectRatio();
});

// Crop the image to a specific size
$image->crop(500, 500);

// Apply a filter to the image (e.g., greyscale)
$image->greyscale();

// Add a watermark to the image
$watermark = Image::make(storage_path('app/public/watermark.png'));
$image->insert($watermark, 'bottom-right', 10, 10);

// Save the manipulated image
$image->save(storage_path('app/public/images/sample_modified.jpg'));


// Apply a filter to the image
$image->filter(function ($filter) {
    // Apply the filter logic
    $filter->sepia();
});
